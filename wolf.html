function draw() { ctx.fillStyle = "#000"; ctx.fillRect(0, 0, canvas.width, canvas.height/2); ctx.fillStyle = "#111"; ctx.fillRect(0, canvas.height/2, canvas.width, canvas.height/2); for (let x = 0; x < canvas.width; x++) { let rayAngle = (player.dir - FOV/2) + (x/canvas.width) * FOV; let dist = 0; let cosA = Math.cos(rayAngle), sinA = Math.sin(rayAngle); while (dist < 20) { dist += 0.05; let tx = Math.floor(player.x + cosA * dist); let ty = Math.floor(player.y + sinA * dist); if (MAP[ty]?.[tx] > 0) break; } let realDist = dist * Math.cos(rayAngle - player.dir); let h = canvas.height / realDist; let y0 = (canvas.height/2) - (h/2); let tx = Math.floor(player.x + cosA * dist); let ty = Math.floor(player.y + sinA * dist); let cell = MAP[ty]?.[tx]; if (cell === 3) { ctx.fillStyle = "gold"; } else { // --- UPDATED DISTANCE SHADER --- // Increased the multiplier (from 12 to 25) to make it darker much sooner. // Adjust the 25 to a higher number (like 40) for even thicker darkness. let brightness = Math.max(0, 255 - (realDist * 25)); ctx.fillStyle = `rgb(0, ${brightness}, 0)`; } ctx.fillRect(x, y0, 1, h); } }
